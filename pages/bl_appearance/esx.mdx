    local function mergeTables(...)
        local mergedTable = {}
        for _, tbl in ipairs({...}) do
            if type(tbl) == "table" then
                for index, value in pairs(tbl) do
                    mergedTable[index] = value
                end
            end
        end
        return mergedTable
    end
